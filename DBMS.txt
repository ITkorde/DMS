
1)Perform the following operations on the database: ALTER TABLE, Pattern Matching using LIKE ,IN,BETWEEN and  Sorting 
	a)Alter table
		CREATE TABLE Student(
  	  	RollNo INT,
   		 Name VARCHAR(30),
    		City VARCHAR(20));

		ALTER TABLE Student ADD Marks INT;

	b) Pattern matching
	-- LIKE
		SELECT * FROM Student
		WHERE Name LIKE 'A%';

	-- IN
		SELECT * FROM Student
		WHERE City IN ('Nagpur','Pune');

	-- BETWEEN
		SELECT * FROM Student
		WHERE Marks BETWEEN 60 AND 90;
	c) Sorting
		SELECT * FROM Student ORDER BY Marks DESC;

2) Operations on the database:Relational Operations (=, <,> ,<= ,>=,AND,OR) , Nested Queries , DROP TABLE 
	a)Relational operator
		SELECT * FROM Student WHERE Marks >= 60 AND City='Nagpur';

		SELECT * FROM Student WHERE Marks < 40 OR City='Pune';

	b)Nested query
		SELECT Name FROM Student
		WHERE Marks = (SELECT MAX(Marks) FROM Student);
	c)Drop table
		DROP TABLE Student;

3)Aggregate Functions using GROUP BY and HAVING Clauses  and Nested Queries  
	a) Aggregate Functions with GROUP BY and HAVING
 		SELECT City, AVG(Marks) FROM Student
		GROUP BY City HAVING AVG(Marks) > 60;

	b) SELECT Name
		FROM Student WHERE Marks > (
   		SELECT AVG(Marks) FROM Student);

4) Create , Drop view and sorting 
	a) Create view 
		CREATE VIEW StudentView AS
		SELECT Name, Marks FROM Student;
	
	b)Drop view 
		DROP VIEW StudentView;
	c) sorting 
		SELECT * FROM Student ORDER BY Name ASC;
5) Create tables with: primary and foreign key
	CREATE TABLE Department(DeptID INT PRIMARY KEY, DeptName VARCHAR(30));

	CREATE TABLE Employee( EmpID INT PRIMARY KEY, EmpName VARCHAR(30), DeptID INT, FOREIGN KEY(DeptID) REFERENCES Department(DeptID));

6) Constraints
	CREATE TABLE Customer(

    	Email VARCHAR(40) UNIQUE,
