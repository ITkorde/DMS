1) Stored functions
 	a)
	CREATE OR REPLACE FUNCTION SquareNum(n NUMBER)
	RETURN NUMBER
	IS
	BEGIN
   		RETURN n*n;
	END;
	/

	SELECT SquareNum(5) FROM dual;
	
	b)
	CREATE OR REPLACE FUNCTION AddNum(a NUMBER,b NUMBER)
	RETURN NUMBER
	IS
	BEGIN
   		RETURN a+b;
	END;
	/

	SELECT AddNum(10,20) FROM dual;
 
	c)
	CREATE OR REPLACE FUNCTION CubeNum(n NUMBER)
	RETURN NUMBER
	IS
	BEGIN
   			RETURN n*n*n;
	END;
	/

	SELECT CubeNum(3) FROM dual;
 
2) Database trigger 
 	a) 
	CREATE OR REPLACE TRIGGER DisplayMsg
	BEFORE INSERT ON Student
	FOR EACH ROW
	BEGIN
  		 DBMS_OUTPUT.PUT_LINE('Record Inserted');
	END;
	/

	b)
	CREATE OR REPLACE TRIGGER SalaryCheck
	BEFORE INSERT ON Employee
	FOR EACH ROW
	BEGIN
   		IF :NEW.EmpID <= 0 THEN
        		RAISE_APPLICATION_ERROR(-20001,'Invalid ID');
   		END IF;
	END;
	/
